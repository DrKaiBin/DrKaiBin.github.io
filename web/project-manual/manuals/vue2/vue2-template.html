# 注册Vue-Router路由管理器

  • 执行安装命令：yarn add vue-router@3.5.3

  • 创建路由

import Vue from 'vue'
import Router from 'vue-router'
// 模块化机制编程，导入Vue和VueRouter，要调用 Vue.use(VueRouter)
Vue.use(Router)

// 路由列表
const routes = [
    {
        path: '/',
        name: '主页',
        component: () => import('@/views/dashboard/index')
    }
]

// 创建路由
const createRouter = () => new Router({
    scrollbarBehavior: () => ({ y: 0 }),
    routes: routes
})

const router = createRouter()

// 路由重置，常用于权限切换，改变路由
export function resetRouter() {
    const newMatchRouter = createRouter()
    router.matcher = newMatchRouter.matcher
}

export default router

# axios全局接口请求封装

  • 通常在三个地方
    1. 初始化axios实例 通常在此设置请求基础路径baseURL、请求超时timeout
    2. 请求拦截 通常在此会设置动作封装如loading、cancelToken，或者对请求参数进行修改、或配置请求头
    3. 响应拦截 通常在此会设置响应后数据返回、错误响应状态等
